```typescript
// Cleaner Engine Interface
interface ICleaner {
  name: string;
  category: CleanerCategory;
  description: string;
  estimatedSize: number;
  safetyLevel: SafetyLevel;
  scan(): Promise<CleanerResult>;
  clean(items: CleanItem[]): Promise<CleanResult>;
}

// Main Cleaner Engine
class CleanerEngine {
  private cleaners: Map<string, ICleaner> = new Map();
  private scanResults: ScanResult[] = [];
  private backupManager: BackupManager;
  private reportGenerator: ReportGenerator;
  
  constructor() {
    this.initializeCleaners();
    this.backupManager = new BackupManager();
    this.reportGenerator = new ReportGenerator();
  }
  
  private initializeCleaners() {
    // Register all cleaners
    this.cleaners.set('temp', new TempFileCleaner());
    this.cleaners.set('browser', new BrowserCleaner());
    this.cleaners.set('recyclebin', new RecycleBinCleaner());
    this.cleaners.set('logs', new LogsCleaner());
    this.cleaners.set('prefetch', new PrefetchCleaner());
    this.cleaners.set('thumbnail', new ThumbnailCleaner());
    this.cleaners.set('duplicate', new DuplicateFileCleaner());
    this.cleaners.set('emptyfolders', new EmptyFolderCleaner());
    this.cleaners.set('registry', new RegistryCleaner());
    this.cleaners.set('windowsupdate', new WindowsUpdateCleaner());
    this.cleaners.set('appcache', new AppCacheCleaner());
    this.cleaners.set('network', new NetworkCleaner());
    this.cleaners.set('fontcache', new FontCacheCleaner());
    this.cleaners.set('driverstore', new DriverStoreCleaner());
    this.cleaners.set('appx', new AppxCleaner());
  }
  
  async scan(selectedCleaners: string[]): Promise<ScanResult> {
    const results: ScanResultItem[] = [];
    let totalSize = 0;
    
    // Create restore point
    await this.backupManager.createRestorePoint('Pre-scan backup');
    
    // Run selected cleaners
    for (const cleanerId of selectedCleaners) {
      const cleaner = this.cleaners.get(cleanerId);
      if (cleaner) {
        const result = await cleaner.scan();
        results.push({
          cleaner: cleaner.name,
          items: result.items,
          size: result.totalSize,
          safety: cleaner.safetyLevel
        });
        totalSize += result.totalSize;
      }
    }
    
    this.scanResults = results;
    
    return {
      items: results,
      totalSize,
      timestamp: new Date(),
      categories: results.length
    };
  }
  
  async clean(selectedItems: CleanItem[]): Promise<CleanResult> {
    // Backup before cleaning
    const backupId = await this.backupManager.backupItems(selectedItems);
    
    const cleaned: CleanItem[] = [];
    const failed: CleanItem[] = [];
    let totalFreed = 0;
    
    // Group by cleaner for efficiency
    const grouped = this.groupByCleaner(selectedItems);
    
    for (const [cleanerId, items] of grouped) {
      const cleaner = this.cleaners.get(cleanerId);
      if (cleaner) {
        try {
          const result = await cleaner.clean(items);
          cleaned.push(...result.successful);
          failed.push(...result.failed);
          totalFreed += result.totalSize;
        } catch (error) {
          console.error(`Cleaner ${cleanerId} failed:`, error);
          failed.push(...items);
        }
      }
    }
    
    // Generate report
    const report = await this.reportGenerator.generate({
      cleaned,
      failed,
      totalFreed,
      backupId,
      timestamp: new Date()
    });
    
    // Save report
    await this.reportGenerator.save(report);
    
    return {
      cleaned,
      failed,
      totalFreed,
      backupId,
      report
    };
  }
  
  async undo(backupId: string): Promise<boolean> {
    return this.backupManager.restore(backupId);
  }
  
  private groupByCleaner(items: CleanItem[]): Map<string, CleanItem[]> {
    const grouped = new Map();
    for (const item of items) {
      if (!grouped.has(item.cleanerId)) {
        grouped.set(item.cleanerId, []);
      }
      grouped.get(item.cleanerId).push(item);
    }
    return grouped;
  }
}
```

### **Backup Manager:**

```typescript
class BackupManager {
  private backupPath: string;
  
  constructor() {
    this.backupPath = path.join(process.env.APPDATA, 'NeoOptimize', 'Backups');
    this.ensureBackupFolder();
  }
  
  async createRestorePoint(description: string): Promise<string> {
    const backupId = generateId();
    const restorePath = path.join(this.backupPath, backupId);
    
    // Create system restore point
    const { exec } = require('child_process');
    await exec(`powershell -Command "Checkpoint-Computer -Description '${description}' -RestorePointType MODIFY_SETTINGS"`);
    
    // Backup registry
    await this.backupRegistry(restorePath);
    
    // Save metadata
    await this.saveMetadata(restorePath, {
      id: backupId,
      description,
      timestamp: new Date(),
      type: 'restore-point'
    });
    
    return backupId;
  }
  
  async backupItems(items: CleanItem[]): Promise<string> {
    const backupId = generateId();
    const itemBackupPath = path.join(this.backupPath, backupId);
    
    await fs.promises.mkdir(itemBackupPath, { recursive: true });
    
    for (const item of items) {
      try {
        // Copy to backup folder
        const destPath = path.join(itemBackupPath, path.basename(item.path));
        await fs.promises.copyFile(item.path, destPath);
        
        // Save metadata
        await this.appendMetadata(itemBackupPath, {
          originalPath: item.path,
          backupPath: destPath,
          size: item.size,
          timestamp: new Date()
        });
      } catch (error) {
        console.error(`Failed to backup ${item.path}:`, error);
      }
    }
    
    return backupId;
  }
  
  async restore(backupId: string): Promise<boolean> {
    const backupPath = path.join(this.backupPath, backupId);
    
    if (!await fs.promises.access(backupPath).then(() => true).catch(() => false)) {
      throw new Error('Backup not found');
    }
    
    // Read metadata
    const metadata = await this.readMetadata(backupPath);
    
    // Restore each file
    for (const item of metadata) {
      try {
        await fs.promises.copyFile(item.backupPath, item.originalPath);
      } catch (error) {
        console.error(`Failed to restore ${item.originalPath}:`, error);
      }
    }
    
    return true;
  }
  
  private async backupRegistry(backupPath: string): Promise<void> {
    const regBackup = path.join(backupPath, 'registry.reg');
    const { exec } = require('child_process');
    await exec(`reg export HKLM\\SOFTWARE "${regBackup}" /y`);
  }
  
  private async ensureBackupFolder(): Promise<void> {
    try {
      await fs.promises.access(this.backupPath);
    } catch {
      await fs.promises.mkdir(this.backupPath, { recursive: true });
    }
  }
  
  private async saveMetadata(backupPath: string, metadata: any): Promise<void> {
    await fs.promises.writeFile(
      path.join(backupPath, 'metadata.json'),
      JSON.stringify(metadata, null, 2)
    );
  }
  
  private async readMetadata(backupPath: string): Promise<any> {
    const data = await fs.promises.readFile(
      path.join(backupPath, 'metadata.json'),
      'utf8'
    );
    return JSON.parse(data);
  }
  
  private async appendMetadata(backupPath: string, item: any): Promise<void> {
    const metadataPath = path.join(backupPath, 'items.json');
    let items = [];
    
    try {
      const data = await fs.promises.readFile(metadataPath, 'utf8');
      items = JSON.parse(data);
    } catch {
      // File doesn't exist yet
    }
    
    items.push(item);
    await fs.promises.writeFile(metadataPath, JSON.stringify(items, null, 2));
  }
}
```
